```sql
-- ── Phase 3: Lists, Columns & Cells ──────────────────────────────────────────

-- Lists — one per team/coach usage; has visibility state
CREATE TABLE IF NOT EXISTS lists (
    id          SERIAL PRIMARY KEY,
    team_id     INTEGER NOT NULL REFERENCES teams(id) ON DELETE CASCADE,
    name        VARCHAR(100) NOT NULL,
    visibility  VARCHAR(10)  NOT NULL DEFAULT 'public'
                CHECK (visibility IN ('public', 'protected', 'private')),
    created_at  TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    updated_at  TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_lists_team_id   ON lists(team_id);
CREATE INDEX IF NOT EXISTS idx_lists_visibility ON lists(visibility);

-- Columns — attribute metadata for EAV (global: list_id IS NULL; local: list_id IS NOT NULL)
-- Global columns: data_type IN ('boolean', 'number') only — text NOT allowed for global (STAT requirement)
-- Local columns: data_type IN ('boolean', 'number', 'text')
CREATE TABLE IF NOT EXISTS columns (
    id          SERIAL PRIMARY KEY,
    team_id     INTEGER NOT NULL REFERENCES teams(id) ON DELETE CASCADE,
    list_id     INTEGER REFERENCES lists(id) ON DELETE CASCADE,
    -- list_id IS NULL => global column (belongs to team); list_id IS NOT NULL => local column (belongs to list)
    name        VARCHAR(100) NOT NULL,
    data_type   VARCHAR(10)  NOT NULL
                CHECK (data_type IN ('boolean', 'number', 'text')),
    -- Application layer must enforce: data_type='text' only when list_id IS NOT NULL
    is_active   BOOLEAN      NOT NULL DEFAULT TRUE,
    sort_order  INTEGER      NOT NULL DEFAULT 0,
    created_at  TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_columns_team_id  ON columns(team_id);
CREATE INDEX IF NOT EXISTS idx_columns_list_id  ON columns(list_id);

-- Cells — EAV values (value stored as TEXT; parsed by app layer per column.data_type)
CREATE TABLE IF NOT EXISTS cells (
    id          SERIAL PRIMARY KEY,
    list_id     INTEGER NOT NULL REFERENCES lists(id)   ON DELETE CASCADE,
    column_id   INTEGER NOT NULL REFERENCES columns(id) ON DELETE CASCADE,
    player_id   INTEGER NOT NULL REFERENCES users(id)   ON DELETE CASCADE,
    value       TEXT,
    created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at  TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE (list_id, column_id, player_id)
);
CREATE INDEX IF NOT EXISTS idx_cells_list_id    ON cells(list_id);
CREATE INDEX IF NOT EXISTS idx_cells_column_id  ON cells(column_id);
CREATE INDEX IF NOT EXISTS idx_cells_player_id  ON cells(player_id);
```

- The `columns` table uses `list_id IS NULL` as the flag for global columns — this is intentional, do not add a separate `is_global` boolean

Key design constraints:
- Functions use `set_admin_context()` for the lookup, then `reset_rls_context()` immediately after — this avoids leaving admin context active
- Functions read `$_SESSION['team_id']`, `$_SESSION['user_id']`, `$_SESSION['role']` from session
- `can_view_list()`: coaches see all lists in their team; players see only public lists
- `can_edit_cell()`: coaches can edit in public/protected lists; players can edit only their own cells in public lists; private lists are never editable by anyone except coaches

```php
<?php
// src/db/visibility.php — Visibility and ownership check helpers for Phase 3
// Used by ALL handlers that touch lists, columns, or cells.
// Application-layer authorization: RLS is defense-in-depth; these checks are authoritative.

declare(strict_types=1);

/**
 * Check if the current session user can VIEW a list (access the list detail page).
 *
 * Rules:
 * - Coaches can view any list belonging to their team (public, protected, private)
 * - Players can view only public lists belonging to their team
 *
 * @param int $list_id  The list to check
 * @return bool
 */
function can_view_list(int $list_id): bool {
    if (empty($_SESSION['team_id']) || empty($_SESSION['role'])) {
        return false;
    }

    $pdo = get_db();
    // Temporarily use admin context to read list metadata without RLS filtering
    reset_rls_context($pdo);
    set_admin_context($pdo);

    $stmt = $pdo->prepare(
        "SELECT visibility, team_id FROM lists WHERE id = ?"
    );
    $stmt->execute([$list_id]);
    $list = $stmt->fetch(PDO::FETCH_ASSOC);

    // Restore team context immediately
    reset_rls_context($pdo);
    set_team_context($pdo, (int)$_SESSION['team_id']);

    if (!$list) {
        return false; // List not found
    }

    // Cross-team access — never allowed
    if ((int)$list['team_id'] !== (int)$_SESSION['team_id']) {
        return false;
    }

    $role = $_SESSION['role'] ?? '';

    if ($role === 'coach') {
        return true; // Coaches see all lists in their team
    }

    if ($role === 'player') {
        return $list['visibility'] === 'public'; // Players see only public lists
    }

    return false;
}

/**
 * Check if the current session user can EDIT a specific player's cells in a list.
 *
 * Rules:
 * - Coaches can edit any player's cells in public or protected lists
 * - Players can edit only their OWN cells, and only in public lists
 * - Private lists: players cannot edit; coaches can edit
 *
 * @param int $list_id    The list containing the cells
 * @param int $player_id  The player whose cells are being edited
 * @return bool
 */
function can_edit_cell(int $list_id, int $player_id): bool {
    if (empty($_SESSION['team_id']) || empty($_SESSION['role']) || empty($_SESSION['user_id'])) {
        return false;
    }

    $pdo = get_db();
    // Temporarily use admin context to read list metadata
    reset_rls_context($pdo);
    set_admin_context($pdo);

    $stmt = $pdo->prepare(
        "SELECT visibility, team_id FROM lists WHERE id = ?"
    );
    $stmt->execute([$list_id]);
    $list = $stmt->fetch(PDO::FETCH_ASSOC);

    // Restore team context immediately
    reset_rls_context($pdo);
    set_team_context($pdo, (int)$_SESSION['team_id']);

    if (!$list) {
        return false;
    }

    // Cross-team access — never allowed
    if ((int)$list['team_id'] !== (int)$_SESSION['team_id']) {
        return false;
    }

    $role       = $_SESSION['role'] ?? '';
    $visibility = $list['visibility'];

    if ($role === 'coach') {
        // Coaches can edit in public and protected lists; NOT in private? No — coaches CAN edit private lists per CELL-03
        return true; // Coaches have full write access to all lists in their team
    }

    if ($role === 'player') {
        // CELL-01: players can only edit their own row in public lists
        $is_own_row   = (int)$_SESSION['user_id'] === $player_id;
        $is_public    = $visibility === 'public';
        return $is_own_row && $is_public;
    }

    return false;
}
```
