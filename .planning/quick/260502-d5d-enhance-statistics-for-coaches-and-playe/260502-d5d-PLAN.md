**Ranking query** — single query using conditional CASE/WHEN inside SUM:

```sql
SELECT
    u.id           AS player_id,
    u.first_name,
    u.last_name,
    c.id           AS column_id,
    c.name         AS column_name,
    c.data_type,
    c.sort_order,

    -- Gesamt: all cells (respects existing date filters from the stats table above)
    COALESCE(
        CASE
            WHEN c.data_type = 'number'  THEN SUM(CASE WHEN (cells.id IS NULL OR <date_filter_condition>) THEN CAST(cells.value AS NUMERIC) ELSE 0 END)
            WHEN c.data_type = 'boolean' THEN SUM(CASE WHEN (cells.id IS NULL OR <date_filter_condition>) THEN CASE WHEN cells.value IN ('true','1') THEN 1 ELSE 0 END ELSE 0 END)
        END, 0
    ) AS sum_all,

    -- Letzte 4 Wochen: lists.date BETWEEN (CURRENT_DATE - INTERVAL '28 days') AND CURRENT_DATE
    COALESCE(
        CASE
            WHEN c.data_type = 'number'  THEN SUM(CASE WHEN lists.date IS NOT NULL AND lists.date >= CURRENT_DATE - INTERVAL '28 days' AND lists.date <= CURRENT_DATE THEN CAST(cells.value AS NUMERIC) ELSE 0 END)
            WHEN c.data_type = 'boolean' THEN SUM(CASE WHEN lists.date IS NOT NULL AND lists.date >= CURRENT_DATE - INTERVAL '28 days' AND lists.date <= CURRENT_DATE THEN CASE WHEN cells.value IN ('true','1') THEN 1 ELSE 0 END ELSE 0 END)
        END, 0
    ) AS sum_4w,

    -- 4–8 Wochen: lists.date BETWEEN (CURRENT_DATE - INTERVAL '56 days') AND (CURRENT_DATE - INTERVAL '29 days')
    COALESCE(
        CASE
            WHEN c.data_type = 'number'  THEN SUM(CASE WHEN lists.date IS NOT NULL AND lists.date >= CURRENT_DATE - INTERVAL '56 days' AND lists.date < CURRENT_DATE - INTERVAL '28 days' THEN CAST(cells.value AS NUMERIC) ELSE 0 END)
            WHEN c.data_type = 'boolean' THEN SUM(CASE WHEN lists.date IS NOT NULL AND lists.date >= CURRENT_DATE - INTERVAL '56 days' AND lists.date < CURRENT_DATE - INTERVAL '28 days' THEN CASE WHEN cells.value IN ('true','1') THEN 1 ELSE 0 END ELSE 0 END)
        END, 0
    ) AS sum_4_8w,

    -- 8–12 Wochen: lists.date BETWEEN (CURRENT_DATE - INTERVAL '84 days') AND (CURRENT_DATE - INTERVAL '57 days')
    COALESCE(
        CASE
            WHEN c.data_type = 'number'  THEN SUM(CASE WHEN lists.date IS NOT NULL AND lists.date >= CURRENT_DATE - INTERVAL '84 days' AND lists.date < CURRENT_DATE - INTERVAL '56 days' THEN CAST(cells.value AS NUMERIC) ELSE 0 END)
            WHEN c.data_type = 'boolean' THEN SUM(CASE WHEN lists.date IS NOT NULL AND lists.date >= CURRENT_DATE - INTERVAL '84 days' AND lists.date < CURRENT_DATE - INTERVAL '56 days' THEN CASE WHEN cells.value IN ('true','1') THEN 1 ELSE 0 END ELSE 0 END)
        END, 0
    ) AS sum_8_12w

FROM users u
CROSS JOIN (
    SELECT id, name, data_type, sort_order
    FROM columns
    WHERE team_id = ? AND list_id IS NULL AND is_active = TRUE
) c
LEFT JOIN cells ON cells.player_id = u.id AND cells.column_id = c.id
LEFT JOIN lists ON cells.list_id = lists.id
WHERE u.team_id = ?
  AND u.role = 'player'
  AND u.is_active = TRUE
  [AND cells.list_id = ? if list_id filter active]
GROUP BY u.id, u.first_name, u.last_name, c.id, c.name, c.data_type, c.sort_order
ORDER BY u.last_name, u.first_name, c.sort_order
```

Build the sort link helper inline in PHP:
```php
// Helper: build URL for sorting by given col_id + win
function ranking_sort_url(int $col_id, string $win, array $current_get): string {
    $params = array_filter([
        'list_id'         => $current_get['list_id'] ?? '',
        'date_from'       => $current_get['date_from'] ?? '',
        'date_to'         => $current_get['date_to'] ?? '',
        'include_undated' => ($current_get['include_undated'] ?? '') ? '1' : '',
        'sort_col'        => $col_id,
        'sort_win'        => $win,
    ], fn($v) => $v !== '');
    return '/coach/stats?' . http_build_query($params);
}
```
